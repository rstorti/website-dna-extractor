    function runExtractionPipeline() {

        if(!input) return;
        
        btn.disabled = true;
        document.getElementById('dashboard').classList.add('hidden');
        document.getElementById('loading-state').classList.remove('hidden');
        
        const text = document.getElementById('loading-text');
        const subtext = document.getElementById('loading-subtext');
        
        setTimeout(() => { text.innerText = 'Analyzing with Gemini Pro...'; subtext.innerText = 'Parsing JSON payload to extract descriptive brand DNA'; }, 1500);
        setTimeout(() => { text.innerText = 'Vertex AI: Imagen 3 Generation...'; subtext.innerText = 'Synthesizing Prompt A (1:1 Aspect Ratio)'; }, 3000);
        setTimeout(() => { subtext.innerText = 'Synthesizing Prompt B (Waiting 2.5s for rate limits)'; }, 4500);
        setTimeout(() => { text.innerText = 'Autonomous Vision Engine...'; subtext.innerText = 'Computing mathematical entropy via sharp.stats() for text placement'; }, 6500);
        setTimeout(() => {
            document.getElementById('loading-state').classList.add('hidden');
            document.getElementById('dashboard').classList.remove('hidden');
            text.innerText = 'Scraping DOM (Headless Chrome)...';
            subtext.innerText = 'Initializing Puppeteer stealth navigation';
            btn.disabled = false;

            list.insertBefore(item, list.firstChild);
        }, 8500);
